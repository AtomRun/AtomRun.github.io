归并排序模板

2020-12-22 #算法模板 
        if (l >= r) {
            return;
        }
        int mid = l + r >> 1;
        merge_sort(q, l, mid);
        merge_sort(q, mid + 1, r);

        int k = 0, i = l, j = mid + 1;
        while (i <= mid && j <= r) {
            if (q[i] <= q[j]) {
                temp[k++] = q[i++];
            } else {
                temp[k++] = q[j++];
            }
        }
        while (i <= mid) {
            temp[k++] = q[i++];
        }
        while (j <= r) {
            temp[k++] = q[i++];
        }
        for (i = l, j = 0; i <= r; i++, j++) {
            q[i] = temp[j];
        }
